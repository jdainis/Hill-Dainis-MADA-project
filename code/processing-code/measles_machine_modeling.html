#Create an object while will run a pipeline that completes a random forest machine model, tunes the random forest model, cross-validates and fits the best RF model to the data based on Mean Absolute Error. 
run_rf_model <- function(data) {
  
  set.seed(1234)
  
  split <- initial_split(data, prop = 0.8)
  train <- training(split)
  test  <- testing(split)

  recipe_obj <- recipe(total_cases_2024_26 ~ ., data = train) %>%
    step_impute_median(all_numeric_predictors()) #Create a recipe for the total 2024-26 measles cases
  
  model <- rand_forest(
    trees = 500,
    mtry  = tune(),
    min_n = tune()
  ) %>%
    set_engine("ranger", importance = "permutation") %>%
    set_mode("regression")#Random Forest Model
  
  wf <- workflow() %>%
    add_model(model) %>%
    add_recipe(recipe_obj)#Create workflow using the random forest model and the recipe
  
  folds <- vfold_cv(train, v = 5)#Cross-validate using folds
  
  grid <- grid_regular(
    mtry(range = c(1, max(1, ncol(train) - 1))),
    min_n(range = c(2, 20)),
    levels = 5
  )#Create a grid to tune off of for the tune_grid() step
  
  tune_res <- tune_grid(
    wf,
    resamples = folds,
    grid = grid,
    metrics = metric_set(mae)
  )#Tune the random forest model using the grid with an output metric of Mean Absolute Error
  
  best <- select_best(tune_res, metric = "mae")#Select the model with the best performing MAE
  
  final_wf <- finalize_workflow(wf, best)#Finalize the workflow
  
  final_fit <- fit(final_wf, data = train)#Fit the data to the final workflow
  
  preds <- predict(final_fit, test) %>%
    bind_cols(test)#Generate predictions on the testing data
  
  metrics <- metrics(preds,
                      truth = total_cases_2024_26,
                      estimate = .pred)#Compare the predictions to the original total_cases_2024_26 data
  
  tibble(
    mae = metrics %>% filter(.metric == "mae") %>% pull(.estimate)
  )#Extract the MAE metrics and place into a tibble
}

#Generate Linear Model to Compare the Random Forest Model to

run_lm_model <- function(data) {
  
  set.seed(1234)
  
  split <- initial_split(data, prop = 0.8)
  train <- training(split)
  test  <- testing(split)
  
  recipe_obj <- recipe(total_cases_2024_26 ~ ., data = train) %>%
    step_impute_median(all_numeric_predictors())
  
  model <- linear_reg() %>%
    set_engine("lm") %>%
    set_mode("regression")
  
  wf <- workflow() %>%
    add_model(model) %>%
    add_recipe(recipe_obj)
  
  fit_model <- fit(wf, data = train)
  
  preds <- predict(fit_model, test) %>%
    bind_cols(test)
  
  metrics <- metrics(preds,
                     truth = total_cases_2024_26,
                     estimate = .pred)
  
  tibble(
    mae = metrics %>% filter(.metric == "mae") %>% pull(.estimate)
  )
}

  dataset = c("Full", "Reduced", "Strong-only"),
  data    = list(ml_measles, reduced_data, strong_data)
) %>%
  mutate(
    rf_metrics = map(data, run_rf_model),
    lm_metrics = map(data, run_lm_model)

  rename(
    rf_mae = rf_metrics_mae,
    lm_mae = lm_metrics_mae
  )
